# 绘制图片

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

:::note

配置使用请指定type="image", 函数调用使用drawImage

::: 

### 支持的参数

```ts
export interface LayoutOptions {
  /**
   * 距离左边距离
   */
  x: number;
  /**
   * 距离顶部距离
   */
  y: number;
  /**
   * 宽度
   */
  width?: number;
  /**
   * 宽度
   */
  height?: number;
  /**
   * 内边距
   * @default 0
   * all
   * [horizontal, vertical]
   * [top, right, bottom, left]
   */
  padding?: number | [number, number] | [number, number, number, number];
  radius?: number | `${number}%`;

  /**
   * drawText 使用
   * 是否外层包裹一个Rect
   * @default false
   */
  wrap?: false | Omit<DrawOptions, 'x' | 'y'>;
  /**
   * 背景色
   */
  backgroundColor?: string;

  // /**
  //  * restore 之前执行，可自行加绘
  //  */
  // afterDraw?: () => void;

  /**
   * 校验，false，则不绘制
   */
  if?: boolean | (() => boolean);

  /**
   *
   */

  /**
   * 是否隔离绘制
   *
   * @default true
   * @description
   * 开始前执行save | 结束后执行restore
   */
  isolation?: boolean;
  /**
   * 边框线
   */
  border?: LayoutBorderOptions;
}

export interface DrawImageOptions extends LayoutOptions {
  type: 'image';
  url: string;
}
```

          type: 'image',

          width: 192 * 2,
          height: 108 * 2,
        },
      ]
      
      eDrawInstance.drawByConfig(drawConfig)
    })

      
    return <canvas id="edraw-canvas1" />
  }

    setTimeout(() => {
