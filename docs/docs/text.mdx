# 绘制文本

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

:::note

配置使用请指定type="text", 函数调用使用drawText

::: 


### 支持的参数

| 参数 | 类型 |默认值 | 描述 |
| :------: | :------: | :------: | :------: |
| text | string | - | 文本 |
| wrap? | false, 矩形参数 | - | 是否外层包裹一个矩形 |

以及
```ts
export interface StyleOptions {
  /**
   * 字体
   * @default 'system-ui'
   */
  fontFamily?: string;
  /**
   * 行高
   * @default 1.2
   */
  lineHeight?: number;
  /**
   * 字体大小
   * @default 13
   */
  fontSize?: number;
  /**
   * 字体颜色
   * @default #333
   */
  color?: string;
  /**
   * 对其方式
   * @default left
   */
  textAlign?: CanvasTextAlign;
  /**
   * 对其方式
   */
  textBaseline?: CanvasTextBaseline;
  /**
   * 加粗
   * @default 'normal'
   */
  bold?: string;

  /**
   * 最大宽度
   * 超过最大宽度会自动换行
   */
  maxWidth?: number;
  /**
   * 多行文本省略
   * 根据maxWidth计算超长省略
   * @default 0
   */
  ellipse?: number;
}
```

### 基本用法

<Tabs>
  <TabItem value="options" label="配置式">

```typescript live
  function () {

    setTimeout(() => {

      const eDrawInstance = new EDraw({
        width: 300,
        height: 50,
        dpr: 1,
        el: "#edraw-canvas1",
      })
    
      const drawConfig = [
        {
          type: 'text',
          text: 'Edraw绘制文本',
          x: 0,
          y: 0,
        },
      ]
      
      eDrawInstance.drawByConfig(drawConfig)
    })

      
    return <canvas id="edraw-canvas1" />
  }
```

### wrap加上padding的例子

<Tabs>
<TabItem value="options" label="配置式">

```typescript live
  function () {

    setTimeout(() => {

      const eDrawInstance = new EDraw({
        width: 600,
        height: 150,
        dpr: 1,
        el: "#edraw-canvas3",
      })
    
      const drawConfig = [
        {
          type: 'text',
          text: 'wrap+padding一般用于文案外的背景',
          x: 40,
          y: 40,
          ellipse: 4,
          maxWidth: 600,
          color: '#000',
          fontSize: 30,
          lineHeight: 30,
          wrap: {
            radius: 20,
            backgroundColor: 'rgba(254, 102, 0, 0.1)',
            padding: 10,
            height: 60,
          },
        },
      ]
      
      eDrawInstance.drawByConfig(drawConfig)
    })

      
    return <canvas id="edraw-canvas3" />
  }
```
