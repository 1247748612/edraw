### 介绍

### 使用

```typescript live

function () {

  setTimeout(() => {

    const eDrawInstance = new EDraw({
      width: 300,
      height: 50,
      dpr: 1,
      el: "#edraw-canvas",
    })
  
    const drawConfig = [
      {
        type: 'text',
        text: '这是基本使用方式, 通常使用配置生成',
        x: 0,
        y: 0,
      },
    ]
    
    eDrawInstance.drawByConfig(drawConfig)
  })

    
  return <canvas id="edraw-canvas" />
}
```
